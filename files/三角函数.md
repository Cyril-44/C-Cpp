## **1. **
$$

\begin{align}
\textbf{I: 基本恒等式}\notag \\
  \sin^2 x + \cos^2 x &= 1 \\
  \tan x &= \frac{\sin x}{\cos x} \\
  1 + \tan^2 x &= \sec^2 x \\
  1 + \cot^2 x &= \csc^2 x \\
\hline
\textbf{II: 诱导公式} \notag \\
 (奇变偶不变，符号看象限) \notag \\
  \sin(x+2k\pi) &= \sin x \\ \cos(x+2k\pi) &= \cos x \\ \tan(x+k\pi) &= \tan x
  \\ \hdashline
  \sin(-x) &= -\sin x \\ \cos(-x) &= \cos x \\ \tan(-x) &= -\tan x
  \\ \hdashline
  \sin(\pi - x) &= \sin x \\ \cos(\pi - x) &= -\cos x
  \\ \hdashline
  \sin(\pi + x) &= -\sin x \\ \cos(\pi + x) &= -\cos x 
  \\ \hdashline
  \sin(\frac{\pi}{2}-x)&=\cos x \\
  \cos(\frac{\pi}{2}-x)&=\sin x
  \\ \hdashline
  \sin(\frac{\pi}{2}+x)&=\cos x \\
  \cos(\frac{\pi}{2}+x)&=-\sin x \\
\hline
\textbf{III: 和差公式} \\
    \sin(a\pm b)&=\sin a\cos b\pm\cos a\sin b \\
    \cos(a\pm b)&=\cos a\cos b\mp\sin a\sin b \\
    \tan(a\pm b)&=\frac{\tan a\pm\tan b}{1\mp\tan a\tan b} \\
\hline
\textbf{IV: 二倍角公式} \\
    \sin 2x &= 2\sin x\cos x \\
    \cos 2x &= \cos^2 x - \sin^2 x \\
    &= 1 - 2\sin^2 x \\
    &= 2\cos^2 x - 1 \\
    \tan 2x &= \frac{2\tan x}{1-\tan^2 x} \\
\hline
\textbf{V: 半角公式} \\
    \sin^2\frac{x}{2} &= \frac{1-\cos x}{2} \\
    \cos^2\frac{x}{2} &= \frac{1+\cos x}{2} \\
    \tan\frac{x}{2} &= \frac{\sin x}{1+\cos x} \\
    \tan\frac{x}{2} &= \frac{1-\cos x}{\sin x} \\
    \tan\frac{x}{2} &= \frac{\sin x}{1+\cos x} \\
\end{align}
$$

<!-- ## **5. 半角公式（竞赛常用）** -->
$$
\sin^2\frac{x}{2} = \frac{1-\cos x}{2}
\cos^2\frac{x}{2} = \frac{1+\cos x}{2}

\tan\frac{x}{2} = \frac{\sin x}{1+\cos x}
\tan\frac{x}{2} = \frac{1-\cos x}{\sin x}
\tan\frac{x}{2} = \frac{\sin x}{1+\cos x}
$$

<!-- ## **6. 积化和差** -->
$$
\sin a\sin b = \frac{1}{2}[\cos(a-b)-\cos(a+b)]
\cos a\cos b = \frac{1}{2}[\cos(a-b)+\cos(a+b)]
\sin a\cos b = \frac{1}{2}[\sin(a+b)+\sin(a-b)]
$$

<!-- ## **7. 和差化积** -->
$$
\sin a+\sin b = 2\sin\frac{a+b}{2}\cos\frac{a-b}{2}
\sin a-\sin b = 2\cos\frac{a+b}{2}\sin\frac{a-b}{2}

\cos a+\cos b = 2\cos\frac{a+b}{2}\cos\frac{a-b}{2}
\cos a-\cos b = -2\sin\frac{a+b}{2}\sin\frac{a-b}{2}
$$

<!-- ## **8. 万能公式（竞赛最常用）** -->
$$
\sin x = \frac{2t}{1+t^2},\quad
\cos x = \frac{1-t^2}{1+t^2},\quad
\tan x = \frac{2t}{1-t^2},
\quad t=\tan\frac{x}{2}
$$

<!-- ## **9. 三倍角与多角公式（竞赛常用）** -->
$$
\sin 3x = 3\sin x - 4\sin^3 x
\cos 3x = 4\cos^3 x - 3\cos x
\tan 3x = \frac{3\tan x - \tan^3 x}{1 - 3\tan^2 x}

\sin(nx) = \Im[(\cos x + i\sin x)^n]
\cos(nx) = \Re[(\cos x + i\sin x)^n]
$$

<!-- ## **11. 常用竞赛技巧公式（重点）** -->

<!-- ### **(1) 经典恒等式** -->
$$
\sin a + \sin b + \sin c = 4\sin\frac{a}{2}\sin\frac{b}{2}\sin\frac{c}{2}
\quad\text{(当 } a+b+c=\pi\text{ 时)}
$$

<!-- ### **(2) 常用化简结构** -->
$$
\frac{1-\cos x}{\sin x} = \tan\frac{x}{2}
\qquad
\frac{1+\cos x}{\sin x} = \cot\frac{x}{2}
$$

<!-- ### **(3) 竞赛常用拆角技巧** -->
$$
\sin x = \sin(a+b) = \sin a\cos b + \cos a\sin b
$$
（用于构造角度、拆分难处理角）

<!-- ### **(4) 竞赛常用对称式** -->
$$
\sin a + \sin b = 2\sin\frac{a+b}{2}\cos\frac{a-b}{2}
\cos a + \cos b = 2\cos\frac{a+b}{2}\cos\frac{a-b}{2}
$$

<!-- ### **(5) 竞赛常用积化差技巧** -->
$$
\sin a\cos b - \cos a\sin b = \sin(a-b)
$$

<!-- ## **12. 反三角函数常用恒等式** -->
$$
\arcsin x + \arccos x = \frac{\pi}{2}
\arctan x + \arctan\frac{1}{x} = \frac{\pi}{2}\quad(x>0)
$$
